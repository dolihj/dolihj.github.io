    # Public Key generate

    # Encrypt with public key

    echo ‘My name is Bob and I want to talk to Alice’ > file.txt

    openssl rsautl -encrypt -inkey 02_public.pem -pubin -in file.txt -out file.ssl openssl rsautl -decrypt -inkey 01private.pem -in file.ssl -out decrypted.txt
